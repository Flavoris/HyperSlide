Replace this:
let playerFrame = player.calculateAccumulatedFrame()
let obstacleFrame = obstacle.calculateAccumulatedFrame()

if playerFrame.intersects(obstacleFrame) {
    continue
}

With a simple geometric check:
let distX = abs(player.position.x - obstacle.position.x)
let distY = abs(player.position.y - obstacle.position.y)

let horizontalOverlap = distX < (playerRadius + obstacle.halfWidth)
let verticalOverlap = distY < (playerRadius + obstacle.halfHeight)

if horizontalOverlap && verticalOverlap {
    // This is a real collision; skip near-miss
    continue
}


No SpriteKit traversal.
No node tree walking.
No expensive bounding box calculations.
